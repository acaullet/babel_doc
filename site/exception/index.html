The Exception class

The class itself

#ifndef EXCEPTION_HPP_
#define EXCEPTION_HPP_

#include <iostream>
#include <exception>
#include <string>

class Exception : public std::exception{
public:
    Exception(const std::string &_where, const std::string &_what) : std::exception() {
        where = _where;
        what = _what;
        errorLog();
        exit(0);
    }
    void    errorLog(void){
        std::cerr << "\n[!] Program crashed at " << where << ":\n[!] " << what << std::endl;
    }
private:
    std::string where;
    std::string what;
};

#endif

class Exception : public std::exception

This class is an extension of the std::exception library by inheritence.

Constructor

Exception(const std::string &_where, const std::string &_what) : std::exception();

Arguments:

  • where : the place where there is an error
  • what : the purpose of the error
Usage: This function is the constructor of our class.

Error Function

void errorLog(void);
Usage: This function is called when we need to print on the error output the place and the purpose of an error.
